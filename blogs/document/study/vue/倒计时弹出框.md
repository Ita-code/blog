### 最近项目中需要一个退出倒计时，然而所以然就开始了下面的。直接代码简单直了

```javascript
import { Message } from 'element-ui'
//倒计时函数
var a = null
function coutTime(index) {
    setTimeout(function () {
        if (index == -1) {
            //index s结束后的操作
        } else {
            a.message = `修改密码成功，${index}s后将自动退出`;
            coutTime(--index);
        }
    }, 1000);
};
export function hint(index) {
    a = Message({
        showClose: false,
        message: `修改密码成功，${index}s后将自动退出`,
        type: "error",
        duration: `${index}000`
    });
    index--
    coutTime(index);
}

```
使用的话就

```javascript
import { hint } from '../hint.js'
            hint(3)


```
